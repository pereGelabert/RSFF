6 Iterate the process over all database and compute the FWI value for each record

## create the empty FWI field
EFFISdbSpatial$FWI <- NA

## Iteration
for(i in 1:nrow(EFFISdbSpatial)){
# print(paste0(i,"/",nrow(EFFISdbSpatial)))  
BurntArea <- EFFISdbSpatial[i,]
date <- gsub("-","",paste0(BurntArea$FIREDATE))
FWI <- brick(FWIfiles[grep(date,FWIfiles)])
FWIburntArea <- crop(rotate(FWI),BurntArea)
EFFISdbSpatial[i,]$FWI <- mean(values(FWIburntArea),na.rm=T)
}

7 Define FWI categories

Categories described here

EFFISdbSpatial <- EFFISdbSpatial%>%
  mutate(FWIcat=cut(FWI, 
    breaks=c(-Inf,5.2, 11.2, 21.3,38,50, Inf), 
    labels=c("Very low","Low","Moderate","High","Very High", "Extreme")))

8 Summarise the results as the proportion of ignitions by year and FWI category

summary_FWI <- as_tibble(EFFISdbSpatial)%>%
  group_by(Year,FWIcat)%>%
  summarise(N=n())%>%
  mutate(freq=(N/sum(N))*100)%>%
  drop_na()
## `summarise()` has grouped output by 'Year'. You can override using the `.groups` argument.

9 Plot the results

ggplot(as_tibble(summary_FWI))+
  geom_col(aes(x=Year,y=freq,fill=FWIcat,color=FWIcat), width =0.4)+
  scale_color_manual(name="FWI categories",values=brewer.pal(6, "YlOrRd"))+
  scale_fill_manual(name="FWI categories",values=alpha(brewer.pal(6, "YlOrRd"),0.65))+
